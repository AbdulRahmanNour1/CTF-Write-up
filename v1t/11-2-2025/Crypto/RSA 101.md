from Cryptodome.Util.number import *
n = 31698460634924412577399959706905435239651
e = 65537
c = 23648999580642514140599125257944114844209
# factordb
p = 313846144900241708687128313929756784551
q = 101 # RSA-101 
phi =(p-1)*(q-1) #31384614490024170868712831392975678455000
# m^e = c mod n
d = pow(e,-1,phi)
# gives e*d = 1 mod phi
# so m^e*d = c^d mod n
# so m^1 = c^d mod n
cleartext = pow(c,d,n)
# result is computed modulo n so we can add n several times if it's not readable
for k in range(5):
    res = long_to_bytes(cleartext + k*n)
    print(f"k={k} : {res}")
